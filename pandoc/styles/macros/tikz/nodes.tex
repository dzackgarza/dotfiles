% ---------------------------------------------------------------------------
% Basic Node Styles
% ---------------------------------------------------------------------------
\tikzset{wh/.style={draw, circle, inner sep=2.5pt, fill=white}} % white
\tikzset{swh/.style={draw, circle, inner sep=2pt, fill=white}}  % small white
\tikzset{dwh/.style={draw, circle, inner sep=3.5pt, fill=white}}% double white
\tikzset{bl/.style={draw=black, circle, inner sep=2.5pt, fill=black}} % black

% ---------------------------------------------------------------------------
% TikZiT Node Styles
% ---------------------------------------------------------------------------
\tikzstyle{white node}=[fill=white, tikzit category=nodes, shape=circle,
  draw=black, tikzit fill=white, scale=0.5]
\tikzstyle{black node}=[fill=black, shape=circle, draw=black,
  tikzit category=nodes, tikzit fill=black, scale=0.5]
\tikzstyle{doubled node}=[fill=white, double, tikzit category=nodes,
  shape=circle, draw=black, tikzit fill=white, scale=0.5]
\tikzstyle{starred node}=[fill=red, tikzit category=nodes, shape=star,
  draw=red, tikzit fill=red, cross out, draw, solid, red, inner sep=2pt]
\tikzstyle{two node}=[style=white node]
\tikzstyle{four node}=[style=black node]

% ---------------------------------------------------------------------------
% Vinberg Node Styles
% ---------------------------------------------------------------------------
\tikzstyle{vinberg white node}=[fill=white, tikzit category=nodes,
  shape=circle, draw=black, tikzit fill=white, scale=0.75]
\tikzstyle{vinberg black node}=[fill=black, shape=circle, draw=black,
  tikzit category=nodes, tikzit fill=black, scale=0.75]
\tikzstyle{vinberg doubled node}=[fill=white, double, tikzit category=nodes,
  shape=circle, draw=black, tikzit fill=white, scale=0.75]

% ---------------------------------------------------------------------------
% ADE Node Styles
% ---------------------------------------------------------------------------
\tikzstyle{ADE blue node}=[black node, fill=blue, draw=blue,
  inner sep=5pt, opacity=0.5]
\tikzstyle{ADE black node}=[black node, inner sep=5pt]
\tikzstyle{ADE white node}=[white node, inner sep=5pt]
\tikzstyle{ADE plain edge}=[plain edge, ultra thick]

% ---------------------------------------------------------------------------
% IAS Styles (node-related)
% ---------------------------------------------------------------------------
\tikzstyle{IAS star}=[fill=red!80, shape=star, draw=red, ultra thick,
  scale=0.5, star point ratio=2.5]
\tikzstyle{IAS background}=[cyan!10]
\tikzstyle{IAS surgery background}=[fill=red!40, draw=none]
\tikzstyle{nontoric blowup}=[fill=red, shape=circle, draw=red, scale=0.5]

% ---------------------------------------------------------------------------
% Cusp Node Styles
% ---------------------------------------------------------------------------
\tikzstyle{cusp point}=[fill=white, tikzit category=nodes, shape=circle,
  draw=black, tikzit fill=white, scale=0.5]
\tikzstyle{BB cusp curve}=[rectangle, draw, minimum width=1.5cm]
\tikzstyle{BB cusp point}=[rounded rectangle, draw, minimum width=2cm]

% ---------------------------------------------------------------------------
% Generic Dot Styles
% ---------------------------------------------------------------------------
\tikzstyle{dot}=[circle, draw=black, fill=black!25, inner sep=.4ex]
\tikzstyle{blackdot}=[dot, fill=black!50]
\tikzstyle{whitedot}=[dot, fill=white]
\tikzstyle{reddot}=[dot, fill=red]
\tikzstyle{greendot}=[dot, fill=green]
