% Works in: LaTeX, Pandoc only (NOT Obsidian MathJax)
%

% ================================================================================
% Complex formatting macros from latexmacs_commands.tex
% ================================================================================

% Case expressions
\providecommand{\fourcase}[4]{
\begin{cases}{#1} & {#2} \\ {#3} & {#4}\end{cases} 
}

% Column vectors
\providecommand{\cvec}[2]{{ \begin{bmatrix} {#1} \\ {#2} \end{bmatrix} }}

% 2x2 matrices
\providecommand{\matt}[4]{{ \begin{bmatrix} {#1} & {#2}  \\ {#3} & {#4} \end{bmatrix} }}

% 3x3 matrices
\providecommand{\mattt}[9]{{
\begin{bmatrix}
  {#1} & {#2} & {#3}
\\
  {#4} & {#5} & {#6}
\\
  {#7} & {#8} & {#9}
\end{bmatrix}
}}

% Stacking with custom spacing
\providecommand{\stack}[1]{\mathclap{\substack{ #1 }}}

\providecommand\stacksymbol[3]{
  \mathrel{\stackunder[2pt]{\stackon[4pt]{$#3$}{$\scriptscriptstyle#1$}}{
  $\scriptscriptstyle#2$}}
}

% Text operators
\providecommand{\textoperatorname}[1]{
  \operatorname{\textnormal{#1}}
}
% Additional complex TeX macros from latexmacs.tex

% Normal subgroup inequality (uses \ooalign and \reflectbox)
\providecommand{\trianglerightneq}{\mathrel{\ooalign{\raisebox{-0.5ex}{\reflectbox{\rotatebox{90}{$\nshortmid$}}}\cr$\triangleright$\cr}\mkern-3mu}}
\providecommand{\normalneq}{\mathrel{\reflectbox{$\trianglerightneq$}}}

% Dirac operator (uses \mkern and spacing hacks)
\providecommand{\dirac}[0]{\mkern-3mu \not{ \partial}}

% Hodge star differential (uses \mkern)
\providecommand{\stardstar}[0]{\hodgestar \mathrm{d} \mkern-1mu \hodgestar}

% Sheaf hom (uses \kern for tight spacing)
\providecommand{\HHom}{\mathscr{H}\kern-2pt\operatorname{om}}
\providecommand{\shom}{ {\mathcal{H}}\kern-0.5pt{\operatorname{om}}}
\providecommand{\QHS}[0]{\mathbf{Q}\kern-0.5pt\operatorname{HS}}

% Bold mathbb BGL with \mkern
\providecommand{\BGL}[0]{ \mathbf{B}\mkern-3mu \operatorname{GL}}

% Git quotient with \mkern
\providecommand{\gitquot}{{ \mathbin{/\mkern-6mu/}}}

% Surjection arrow with \mkern
\providecommand{\surjectsvia}[2][]{ \xrightarrow[#1]{#2} { \mathrel{\mkern-16mu}\rightarrow }\, }

% Quotient constructions with \mkern
\providecommand{\quotleft}[2]{ {}_{#2}\mkern-.5mu\backslash\mkern-2mu^{#1} }
\providecommand{\quotright}[2]{ {}^{#1}\mkern-2mu/\mkern-2mu_{#2} }

% Multi-choose using \kern
\def\multichoose#1#2{{\left(\kern-.3em\left(\genfrac{}{}{0pt}{}{#1}{#2}\right)\kern-.3em\right)}}

% Augmentation ideal (uses \fboxsep and \fbox)
\providecommand{\aug}{\fboxsep=-\fboxrule\!\!\!\fbox{\strut}\!\!\!}

% Large symbols using text sizing
\providecommand{\bigast}[0]{{\mathop{\text{\Large $\ast$}}}}
\providecommand{\connectsum}[0]{\mathop{ \Large\mypound }}

% Horizontal and vertical bars using \rule
\providecommand*{\horzbar}{\rule[.5ex]{2.5ex}{0.5pt}}
\providecommand*{\vertbar}{\rule[-1ex]{0.5pt}{2.5ex}}

% Scaled and raised symbols
\providecommand{\mypound}{\scalebox{0.8}{\raisebox{0.4ex}{\#}}}

% Triple arrows using \substack
\providecommand\rrrarrows{
    \mathrel{\substack{\textstyle\rightarrow\\[-0.6ex]
        \textstyle\rightarrow \\[-0.6ex]
        \textstyle\rightarrow}}
}
