\usetikzlibrary{patterns}
\begin{tikzpicture}[scale=1.5]
  % Isometric view of a disc (ellipse)
  \draw[thick] (0,0) ellipse (2cm and 0.8cm);
  
  % Hatching/shading to indicate the disc surface
  \begin{scope}
    \clip (0,0) ellipse (2cm and 0.8cm);
    \fill[pattern=north east lines, pattern color=gray!40] (-2,-1) rectangle (2,1);
  \end{scope}

  % Boundary points and vectors (representative of the 16-gon)
  \foreach \a in {0, 22.5, ..., 337.5} {
    \coordinate (p) at ({\a}:2cm and 0.8cm);
    \fill (p) circle (1.5pt);
    % Representative vectors l_i v_i
    \ifdim \a pt > 180pt
      \draw[->, thick, red!70!black] (p) -- +({\a+10}:0.5cm);
    \else
      \draw[->, thick, red!70!black] (p) -- +({\a-10}:0.5cm);
    \fi
  }

  \node at (0, -1.2) {Disc slice of inverted cone ($B(\lambda)$)};
\end{tikzpicture}
