\usetikzlibrary{patterns}
\begin{tikzpicture}[scale=1.5]
    % Define vertices of the rectangle
    \coordinate (A) at (0, 2);   % Top
    \coordinate (B) at (-2, 0);  % Left
    \coordinate (C) at (0, -2);  % Bottom
    \coordinate (D) at (2, 0);   % Right
    \coordinate (M) at (0, 0);   % Center
    
    % Shading patterns (hatching) - NO INTERNAL LINES
    \fill[pattern=north east lines, pattern color=green!60!black] (A) -- (-1,1) -- (M) -- (1,1) -- cycle;
    \fill[pattern=vertical lines, pattern color=blue!60!black] (B) -- (-1,1) -- (M) -- (-1,-1) -- cycle;
    \fill[pattern=north west lines, pattern color=red!60!black] (C) -- (-1,-1) -- (M) -- (1,-1) -- cycle;
    \fill[pattern=horizontal lines, pattern color=orange!60!black] (D) -- (1,1) -- (M) -- (1,-1) -- cycle;
    
    % Draw the boundary
    \draw[thick] (A) -- (B) -- (C) -- (D) -- cycle;
    
    % Vertices (16-gon boundary has more points)
    \foreach \p in {A,B,C,D} \fill (\p) circle (2pt);
    \foreach \t in {0.33, 0.66} {
        \fill ($(A)!\t!(B)$) circle (1.5pt);
        \fill ($(B)!\t!(C)$) circle (1.5pt);
        \fill ($(C)!\t!(D)$) circle (1.5pt);
        \fill ($(D)!\t!(A)$) circle (1.5pt);
    }
    
    % Symington surgeries: small triangles along the edges pointing inward
    \foreach \pA/\pB in {A/B, B/C, C/D, D/A} {
        \foreach \t in {0.2, 0.5, 0.8} {
            \coordinate (pt) at ($(\pA)!\t!(\pB)$);
            \draw[red, thick] (pt) -- ++(135:0.2) -- ++(-90:0.282) -- cycle; % Small triangle
        }
    }
    
    % Labels
    \draw[->, thick] (-2, -1) -- +(-0.5, -0.5) node[below left] {$(3, -3)$};
    \draw[->, thick] (2, -1) -- +(0.5, 0.5) node[above right] {$(2, 2)$};
\end{tikzpicture}
